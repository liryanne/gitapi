001  (ns gitapi.git
002    (:require [clj-http.client :as http]
003              [clojure.set :refer [rename-keys]]
004              [cheshire.core :as json]
005              [gitapi.sql :as sql]))
006  
007  (defn parse
008    [data]
009    (json/parse-string data (fn [k] (keyword k))))
010  
011  (defn get-http
012    [url]
013    (try
014      (parse (:body (http/get url)))
015      (catch Exception e
016        (println (str "get-http error: " e))
017        nil)))
018  
019  (defn get-user
020    [username]
021    (let [url (str "https://api.github.com/users/" username)
022          user (get-http url)]
023      (-> user
024        (select-keys [:id :repos_url :login])
025        (rename-keys {:login :username}))))
026  
027  (def keys-repo [:id :name :html_url :watchers :size :updated_at :stargazers_count :created_at])
028  
029  (defn format-repositories
030    [repositories]
031    (vec (map #(rename-keys (select-keys % keys-repo) {:html_url :url, :stargazers_count :stargazers}) repositories)))
032  
033  (defn get-repositories
034    [username]
035    (let [{url :repos_url
036           username :username
037           user-id :id} (get-user username)
038          repositories (get-http url)]
039      {:user_id user-id
040       :username username
041       :repositories (format-repositories repositories)}))
042  
043  (defn get-repository
044    [username repository-name save-data]
045    (let [data (get-repositories username)
046          {:keys [user_id username]} data
047          repositories (:repositories data)
048          repository (first (filter #(= (:name %) repository-name) repositories))]
049      (if (true? save-data)
050        (do
051          (sql/insert-user! user_id username)
052          (sql/insert-repository!
053            (:id repository)
054            (:name repository)
055            user_id
056            (:url repository)
057            (:size repository)
058            (:stargazers repository)
059            (:watchers repository)
060            (:updated_at repository)
061            (:created_at repository))))
062      repository))
