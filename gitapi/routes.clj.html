001  (ns gitapi.routes
002    (:require [reitit.ring :as ring]
003              [gitapi.handlers :refer :all]
004              [reitit.ring :as ring]
005              [reitit.ring.middleware.muuntaja :as muuntaja]
006              [reitit.ring.middleware.exception :as exception]
007              [reitit.ring.middleware.muuntaja :as muuntaja]
008              [reitit.ring.middleware.parameters :as parameters]
009              [reitit.ring.coercion :as coercion]
010              [muuntaja.core :as m]
011              [reitit.coercion.spec])
012    (:use [clojure.walk]))
013  
014  (def routes
015    [["/repositories"
016      {:get {:parameters {:query-params {:username string?
017                                         :from_local boolean?}}
018             :handler (fn [{:keys [query-params]}]
019                        (get-repositories-handler (keywordize-keys query-params)))}}]
020  
021     ["/repositories/:username/:repository"
022      {:get {:parameters {:query-params {:save_data boolean?}}
023             :handler (fn [{:keys [path-params query-params]}]
024                        (get-repository-handler (merge path-params (keywordize-keys query-params))))}}]])
025  
026  (def router
027    (ring/router routes
028      {:data {:coercion reitit.coercion.spec/coercion
029              :muuntaja m/instance
030              :middleware [muuntaja/format-negotiate-middleware
031                           muuntaja/format-response-middleware
032                           exception/exception-middleware
033                           muuntaja/format-request-middleware
034                           coercion/coerce-request-middleware
035                           coercion/coerce-response-middleware
036                           parameters/parameters-middleware]}}))
