001  (ns gitapi.sql
002    (:require [gitapi.db :as db]
003              [clojure.set :refer [rename-keys]]))
004  
005  (defn insert-user!
006    ([id username]
007     (first (db/insert-user db/spec {:id id, :username username}))))
008  
009  (defn get-user-by-username
010    [username]
011    (first (db/select-user-by-username db/spec {:username username})))
012  
013  (defn insert-repository!
014    [id name user-id url size stargazers watchers updated-at created-at]
015    (let [repository {:id id
016                      :name name
017                      :user-id user-id
018                      :url url
019                      :size size
020                      :stargazers stargazers
021                      :watchers watchers
022                      :updated-at updated-at
023                      :created-at created-at}]
024      (first (db/insert-repository db/spec repository))))
025  
026  (defn get-repositories-by-user
027    [username]
028    (db/select-repositories-by-user db/spec {:username username}))
029  
030  (defn get-repositories-from-local
031    [username]
032    (let [repositories-db (get-repositories-by-user username)]
033      (if-not (empty? repositories-db)
034        (let [user (get-user-by-username username)]
035          (merge
036            (rename-keys user {:id :user_id})
037            {:repositories (vec (map #(dissoc % :user_id) repositories-db))})))))
