001  (ns gitapi.handlers
002    (:require [gitapi.git :as git]
003              [ring.util.http-response :refer :all]
004              [gitapi.sql :refer [get-repositories-from-local]]
005              [clojure.set :refer [rename-keys]]))
006  
007  (defn get-repositories-handler
008    [{:keys [username from_local]}]
009    (let [from-local (Boolean/valueOf from_local)
010          repositories (if (true? from-local)
011                         (get-repositories-from-local username)
012                         (git/get-repositories username))]
013      (ok (if (empty? repositories)
014            {}
015            repositories))))
016  
017  (defn get-repository-handler
018    [{:keys [username repository save_data]}]
019    (if (nil? save_data)
020      (bad-request {:message "Parameter save_data required"})
021  
022      (let [save-data (Boolean/valueOf save_data)
023            repo (git/get-repository username repository save-data)]
024        (ok repo))))
